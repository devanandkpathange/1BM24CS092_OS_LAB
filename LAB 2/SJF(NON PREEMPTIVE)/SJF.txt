
#include<stdio.h>

struct a
{
    int pid;
    int at;
    int bt;
    int ct;
    int wt;
    int tat;
    int com;
};

int main()
{
    int n,t,cur=0,completed=0,j[100],k,b; double swt=0,stat=0;
    printf("Enter total number of processes: ");
    scanf("%d",&n);
    struct a sjf[n];
    for(int i=0;i<n;i++)
    {
        printf("Enter arrival time, burst time and process id: ");
        scanf("%d%d%d",&sjf[i].at,&sjf[i].bt,&sjf[i].pid);
        sjf[i].com=0;
        printf("\n");
    }
    while(completed<n)
    {
        k=0;
        for(int i=0;i<n;i++)
        {
            if(sjf[i].at<=cur && sjf[i].com!=1)
            {
                j[k++]=i;
            }
        }
        b = j[0];
        for(int a = 1; a < k; a++)
        {
            if(sjf[j[a]].bt < sjf[b].bt)
            {
                b = j[a];
            }
        }
        if(k==0)
        {
            cur++;
        }
        else
        {
            completed++;
            sjf[b].com=1;
            sjf[b].ct=cur+sjf[b].bt;
            sjf[b].tat=sjf[b].ct-sjf[b].at;
            sjf[b].wt=sjf[b].tat-sjf[b].bt;
            cur=sjf[b].ct;
            swt=swt+sjf[b].wt;
            stat=stat+sjf[b].tat;
        }
    }
    swt=swt/n;
    stat=stat/n;
    printf("Process_id \t Arrival_time \t Burst_time \t Completion_time \t Waiting_time \t Turnaround_time \t");
    for(int i=0;i<n;i++)
    {
        printf("\n");
        printf("%d\t\t%d\t\t%d\t\t%d\t\t\t%d\t\t%d\t\t",sjf[i].pid,sjf[i].at,sjf[i].bt,sjf[i].ct,sjf[i].wt,sjf[i].tat);
    }
    printf("\nAverage Waiting time: %lf",swt);
    printf("\nAverage Turnaround time:%lf",stat);
    return 0;
}
