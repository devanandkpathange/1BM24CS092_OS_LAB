#include <stdio.h>
#include <math.h>

typedef struct {
    int id, period, burst, deadline;
    int remaining, arrival;
} Task;

int chooseTask(Task t[], int n) {

    int best = -1;

    for (int i = 0; i < n; i++) {

        if (t[i].remaining > 0) {

            if (best == -1 || t[i].period < t[best].period)
                best = i;
        }
    }

    return best;
}

int schedulable(Task t[], int n) {

    float u = 0, bound;

    for (int i = 0; i < n; i++)
        u += (float)t[i].burst / t[i].period;

    bound = n * (pow(2, 1.0 / n) - 1);

    printf("\nCPU Utilization = %.4f\n", u);
    printf("RMS Bound = %.4f\n", bound);

    return u <= bound;
}

int main() {

    int n, simTime;

    printf("Rate Monotonic Scheduling\n");

    printf("Enter number of tasks: ");
    scanf("%d", &n);

    Task t[n];

    for (int i = 0; i < n; i++) {

        t[i].id = i + 1;

        printf("\nTask T%d Period: ", i + 1);
        scanf("%d", &t[i].period);

        printf("Task T%d Execution Time: ", i + 1);
        scanf("%d", &t[i].burst);

        printf("Task T%d Deadline (enter 0 for same as period): ", i + 1);
        scanf("%d", &t[i].deadline);

        if (t[i].deadline == 0)
            t[i].deadline = t[i].period;

        t[i].remaining = 0;
        t[i].arrival = 0;
    }

    if (schedulable(t, n))
        printf("Schedulable using RMS\n");
    else {
        printf("Not Schedulable using RMS\n");
        return 0;
    }

    printf("\nEnter Simulation Time: ");
    scanf("%d", &simTime);

    printf("\nFrom\tTo\tTask\n");

    int current = -1, start = 0;

    for (int time = 0; time < simTime; time++) {

        for (int i = 0; i < n; i++) {

            if (time == t[i].arrival) {

                t[i].remaining = t[i].burst;
                t[i].arrival += t[i].period;
            }
        }

        int selected = chooseTask(t, n);

        if (selected != -1)
            t[selected].remaining--;

        if (selected != current) {

            if (time != 0) {

                if (current == -1)
                    printf("%4d\t%2d\tIDLE\n", start, time);
                else
                    printf("%4d\t%2d\tT%d\n", start, time, current + 1);
            }

            start = time;
            current = selected;
        }
    }

    if (current == -1)
        printf("%4d\t%2d\tIDLE\n", start, simTime);
    else
        printf("%4d\t%2d\tT%d\n", start, simTime, current + 1);

    printf("\nTask Summary:\n");
    printf("Task\tPeriod\tExec\tDeadline\n");

    for (int i = 0; i < n; i++) {

        printf("T%d\t%d\t%d\t%d\n",
               t[i].id,
               t[i].period,
               t[i].burst,
               t[i].deadline);
    }

    return 0;
}
